# ntpdate, enum4linux-ng, john, hashcat, seclists) and kerbrute (Go binary in
# /opt/kerbrute) are handled by install.sh, not by pip.

bloodhound         # bloodhound-python ingestor
bloodyAD           # fast LDAP privilege/ACL abuse
ldapdomaindump     # offline LDAP dump (optional)
msoffcrypto-tool   # decrypt password-protected Office docs found on shares
